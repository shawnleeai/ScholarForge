```python
class PPTGenerator:
    """AI PPT生成器"""

    def generate(self, paper_content: str) -> PPT:
        # 1. 提取关键信息
        sections = self.extract_sections(paper_content)

        # 2. 生成大纲
        outline = self.llm.generate_outline(sections)

        # 3. 生成每页内容
        slides = []
        for item in outline:
            slide = self.generate_slide(item)
            slides.append(slide)

        # 4. 应用模板
        return self.apply_template(slides, template='academic')
```
