```python
# scheduler.py - 补充实现
class TaskScheduler:
    def __init__(self):
        self.task_queue = TaskQueue()
        self.agents: Dict[str, Agent] = {}
        self.running_tasks: Dict[str, AgentTask] = {}

    async def schedule(self, task: AgentTask) -> Optional[str]:
        # 1. 筛选有能力的Agent
        capable = [a for a in self.agents.values()
                   if self._can_execute(a, task)]

        if not capable:
            return None

        # 2. 计算调度分数
        best_agent = max(capable,
            key=lambda a: self._calculate_score(a, task))

        # 3. 分配任务
        await self._assign(task, best_agent.id)
        return best_agent.id

    def _calculate_score(self, agent: Agent, task: AgentTask) -> float:
        # 能力匹配40% + 负载因子40% + 响应估计20%
        capability = self._match_capability(agent, task)
        load = 1.0 - (agent.current_load / agent.max_capacity)
        response = self._estimate_response(agent, task)
        return capability * 0.4 + load * 0.4 + response * 0.2
```
